<td class="answercell">
    <div class="post-text" itemprop="text">
<p><a href="http://nodejs.org/docs/v0.4.10/api/http.html#http.request">http.request</a> docs contains example how to receive body of the response through handling <code>data</code> event:</p>

<pre><code>var options = {
  host: 'www.google.com',
  port: 80,
  path: '/upload',
  method: 'POST'
};

var req = http.request(options, function(res) {
  console.log('STATUS: ' + res.statusCode);
  console.log('HEADERS: ' + JSON.stringify(res.headers));
  res.setEncoding('utf8');
  res.on('data', function (chunk) {
    console.log('BODY: ' + chunk);
  });
});

req.on('error', function(e) {
  console.log('problem with request: ' + e.message);
});

// write data to request body
req.write('data\n');
req.write('data\n');
req.end();
</code></pre>

<p><a href="http://nodejs.org/docs/v0.4.10/api/http.html#http.get">http.get</a> does the same thing as http.request except it calls <code>req.end()</code> automatically.</p>

<pre><code>var options = {
  host: 'www.google.com',
  port: 80,
  path: '/index.html'
};

http.get(options, function(res) {
  console.log("Got response: " + res.statusCode);

  res.on("data", function(chunk) {
    console.log("BODY: " + chunk);
  });
}).on('error', function(e) {
  console.log("Got error: " + e.message);
});
</code></pre>

<p>I also want to add that the <code>http.ClientResponse</code> returned by <code>http.get()</code> has an <code>end</code> event, so here is another way that I receive the body response:</p>

<pre><code>var options = {
  host: 'www.google.com',
  port: 80,
  path: '/index.html'
};

http.get(options, function(res) {
  var body = '';
  res.on('data', function(chunk) {
    body += chunk;
  });
  res.on('end', function() {
    console.log(body);
  });
}).on('error', function(e) {
  console.log("Got error: " + e.message);
}); 
</code></pre>

<h3>Edit: replying to self 6 years later</h3>

<p>The <a href="https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Operators/await" rel="nofollow noreferrer"><em>await</em> keyword</a> is the best way to handle this, avoiding callbacks and <code>.then()</code></p>

<pre><code>const superagent = require('superagent');

const start = async function(){
  const res = await superagent.get('https://www.google.com')
  console.log(res.text)
}

start();
</code></pre>

<p>Using await, control <strong>simply passes onto the next line</strong> once the promise returned by <code>superagent.get()</code> has a value.</p>

<p>A portion of Coffee here:</p>

<pre><code># My little helper
read_buffer = (buffer, callback) -&gt;
  data = ''
  buffer.on 'readable', -&gt; data += buffer.read().toString()
  buffer.on 'end', -&gt; callback data

# So request looks like
http.get 'http://i.want.some/stuff', (res) -&gt;
  read_buffer res, (response) -&gt;
    # Do some things with your response
    # but don't do that exactly :D
    eval(CoffeeScript.compile response, bare: true)
</code></pre>

<p>And compiled</p>

<pre><code>var read_buffer;

read_buffer = function(buffer, callback) {
  var data;
  data = '';
  buffer.on('readable', function() {
    return data += buffer.read().toString();
  });
  return buffer.on('end', function() {
    return callback(data);
  });
};

http.get('http://i.want.some/stuff', function(res) {
  return read_buffer(res, function(response) {
    return eval(CoffeeScript.compile(response, {
      bare: true
    }));
  });
});
</code></pre>
